01 · IDENTITY

الهوية والمبادئ

سباير ميديكال منصة طبية رقمية في العراق · التركيز الأساسي على سحب الدم المنزلي والتحاليل المختبرية (~90٪ من العمل). التصميم يستلهم من Google Play لتقديم تجربة مألوفة ومريحة للمستخدم العراقي.

المبدأ التطبيق
🎯 وضوح أولاً كل عنصر في الواجهة له سبب وهدف. لا توجد عناصر زخرفية بدون وظيفة.
📱 موبايل أولاً التصميم مبني لشاشة 480px ثم يتوسع للأكبر. 100٪ من المستخدمين على الهاتف.
🩺 ثقة طبية ألوان هادئة · خطوط واضحة · معلومات دقيقة · لا مبالغة بصرية.
🇮🇶 سياق عراقي اللهجة العراقية في النصوص · أسعار بالدولار · توقيت بغداد · مرونة في الدفع.
⚡ سرعة فورية كل صفحة تحمل في أقل من 2 ثانية · skeleton loaders · PWA كامل.
♿ وصول كامل WCAG AA · تباين مرتفع · أحجام مس مناسبة · دعم قارئ الشاشة.

CREATE TABLE admin_stories (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,                    -- "عرض خاص"
  image_url TEXT,                          -- صورة الستوري
  background_color TEXT DEFAULT '#01875F',
  badge_text TEXT,                         -- "15$" أو "-30%"
  action_url TEXT,                         -- /services/blood-test
  display_order INT DEFAULT 0,           -- ترتيب العرض
  starts_at TIMESTAMP DEFAULT NOW(),
  ends_at TIMESTAMP,                       -- صلاحية الستوري
  is_active BOOLEAN DEFAULT true,
  views_count INT DEFAULT 0,
  clicks_count INT DEFAULT 0,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- جدول لتسجيل مشاهدات المستخدمين (لحالة "مشاهد")
CREATE TABLE user_story_views (
  user_id UUID REFERENCES auth.users(id),
  story_id UUID REFERENCES admin_stories(id),
  viewed_at TIMESTAMP DEFAULT NOW(),
  PRIMARY KEY (user_id, story_id)
);

-- RLS · فقط الأدمن يستطيع تعديل
ALTER TABLE admin_stories ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public_read_active" ON admin_stories
  FOR SELECT USING (is_active = true AND (ends_at IS NULL OR ends_at > NOW()));
